26std::shared_ptr<tatami::Matrix<Data_, Index_> > load_sparse_matrix_basic(Parser_& parser, eminem::Field field, size_t NR, size_t NC, size_t NL) {
27 std::vector<typename std::conditional<row_, TempIndex_, StoredIndex_>::type> rows;
28 std::vector<typename std::conditional<!row_, TempIndex_, StoredIndex_>::type> columns;
29 rows.reserve(NL), columns.reserve(NL);
30 std::vector<StoredData_> values;
31 values.reserve(NL);
32
33 if (field == eminem::Field::INTEGER) {
34 parser.scan_integer([&](size_t r, size_t c, int v) -> void {
35 values.push_back(v);
36 rows.push_back(r - 1);
37 columns.push_back(c - 1);
38 });
39
40 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
41 parser.scan_real([&](size_t r, size_t c, double v) -> void {
42 values.push_back(v);
43 rows.push_back(r - 1);
44 columns.push_back(c - 1);
45 });
46
47 } else {
48 throw std::runtime_error("unsupported Matrix Market field type");
49 }
50
51 auto ptr = tatami::compress_sparse_triplets<row_>(NR, NC, values, rows, columns);
52 std::vector<StoredIndex_> indices;
53 if constexpr(row_) {
54 indices.swap(columns);
55 } else {
56 indices.swap(rows);
57 }
58
59 return std::shared_ptr<tatami::Matrix<Data_, Index_> >(
60 new tatami::CompressedSparseMatrix<row_, Data_, Index_, decltype(values), decltype(indices), decltype(ptr)>(
61 NR, NC, std::move(values), std::move(indices), std::move(ptr), false
62 )
63 );

107template<bool row_, typename Data_, typename Index_, typename StoredData_, typename Parser_>
108std::shared_ptr<tatami::Matrix<Data_, Index_> > load_dense_matrix_basic(Parser_& parser, eminem::Field field, size_t NR, size_t NC) {
109 std::vector<StoredData_> values;
110 auto total_size = tatami::multiply(NR, NC);
111 if constexpr(row_) {
112 values.resize(total_size);
113 } else {
114 values.reserve(total_size);
115 }
116
117 if (field == eminem::Field::INTEGER) {
118 parser.scan_integer([&](size_t r, size_t c, int v) -> void {
119 if constexpr(row_) {
120 values[tatami::multiply_and_add(r - 1, NC, c - 1)] = v;
121 } else {
122 values.push_back(v); // Matrix Market ARRAY format is already column-major
123 }
124 });
125
126 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
127 parser.scan_real([&](size_t r, size_t c, double v) -> void {
128 if constexpr(row_) {
129 values[tatami::multiply_and_add(r - 1, NC, c - 1)] = v;
130 } else {
131 values.push_back(v);
132 }
133 });
134
135 } else {
136 throw std::runtime_error("unsupported Matrix Market field type");
137 }
138
139 return std::shared_ptr<tatami::Matrix<Data_, Index_> >(
140 new tatami::DenseMatrix<row_, Data_, Index_, decltype(values)>(NR, NC, std::move(values))
141 );
142}

167std::shared_ptr<tatami::Matrix<Data_, Index_> > load_matrix(byteme::Reader& reader) {
168 eminem::Parser<parallel_> parser(&reader);
169 parser.scan_preamble();
170
171 const auto& banner = parser.get_banner();
172 auto field = banner.field;
173 size_t NR = parser.get_nrows(), NC = parser.get_ncols(), NL = parser.get_nlines();
174
175 if (banner.format == eminem::Format::COORDINATE) {
176 // Automatically choosing a smaller integer type for the temporary index.
177 constexpr size_t limit8 = std::numeric_limits<uint8_t>::max(), limit16 = std::numeric_limits<uint16_t>::max();
178 auto primary = (row_ ? NR : NC);
179
180 if (primary <= limit8) {
181 return load_sparse_matrix_index<row_, Data_, Index_, StoredData_, StoredIndex_, uint8_t>(parser, field, NR, NC, NL);
182 } else if (primary <= limit16) {
183 return load_sparse_matrix_index<row_, Data_, Index_, StoredData_, StoredIndex_, uint16_t>(parser, field, NR, NC, NL);
184 } else {
185 return load_sparse_matrix_index<row_, Data_, Index_, StoredData_, StoredIndex_, uint32_t>(parser, field, NR, NC, NL);
186 }
187
188 } else {
189 if constexpr(std::is_same<StoredData_, Automatic>::value) {
190 if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
191 return load_dense_matrix_basic<row_, Data_, Index_, double>(parser, field, NR, NC);
192 }
193 if (field != eminem::Field::INTEGER) {

195 }
196 return load_dense_matrix_basic<row_, Data_, Index_, int>(parser, field, NR, NC);

199 return load_dense_matrix_basic<row_, Data_, Index_, StoredData_>(parser, field, NR, NC);
