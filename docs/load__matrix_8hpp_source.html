26std::shared_ptr<tatami::Matrix<Data_, Index_> > load_sparse_matrix_basic(Parser_& parser, eminem::Field field, size_t NR, size_t NC, size_t NL) {
27 std::vector<typename std::conditional<row_, TempIndex_, StoredIndex_>::type> rows;
28 std::vector<typename std::conditional<!row_, TempIndex_, StoredIndex_>::type> columns;
29 rows.reserve(NL), columns.reserve(NL);
30 std::vector<StoredData_> values;
31 values.reserve(NL);
32
33 if (field == eminem::Field::INTEGER) {
34 parser.scan_integer([&](size_t r, size_t c, int v) -> void {
35 values.push_back(v);
36 rows.push_back(r - 1);
37 columns.push_back(c - 1);
38 });
39
40 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
41 parser.scan_real([&](size_t r, size_t c, double v) -> void {
42 values.push_back(v);
43 rows.push_back(r - 1);
44 columns.push_back(c - 1);
45 });
46
47 } else {
48 throw std::runtime_error("unsupported Matrix Market field type");
49 }
50
51 auto ptr = tatami::compress_sparse_triplets<row_>(NR, NC, values, rows, columns);
52 std::vector<StoredIndex_> indices;
53 if constexpr(row_) {
54 indices.swap(columns);
55 } else {
56 indices.swap(rows);
57 }
58

60 new tatami::CompressedSparseMatrix<row_, Data_, Index_, decltype(values), decltype(indices), decltype(ptr)>(

108std::shared_ptr<tatami::Matrix<Data_, Index_> > load_dense_matrix_basic(Parser_& parser, eminem::Field field, size_t NR, size_t NC) {
109 std::vector<StoredData_> values;
110 if constexpr(row_) {
111 values.resize(NR * NC);
112 } else {
113 values.reserve(NR * NC);
114 }
115
116 if (field == eminem::Field::INTEGER) {
117 parser.scan_integer([&](size_t r, size_t c, int v) -> void {
118 if constexpr(row_) {
119 values[(r - 1) * NC + (c - 1)] = v;
120 } else {
121 values.push_back(v); // Matrix Market ARRAY format is already column-major
122 }
123 });
124
125 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
126 parser.scan_real([&](size_t r, size_t c, double v) -> void {
127 if constexpr(row_) {
128 values[(r - 1) * NC + (c - 1)] = v;
129 } else {
130 values.push_back(v);
131 }
132 });
133
134 } else {
135 throw std::runtime_error("unsupported Matrix Market field type");
136 }
137
138 return std::shared_ptr<tatami::Matrix<Data_, Index_> >(
139 new tatami::DenseMatrix<row_, Data_, Index_, decltype(values)>(NR, NC, std::move(values))

167 eminem::Parser<parallel_> parser(&reader);
168 parser.scan_preamble();
169
170 const auto& banner = parser.get_banner();
171 auto field = banner.field;
172 size_t NR = parser.get_nrows(), NC = parser.get_ncols(), NL = parser.get_nlines();
173
174 if (banner.format == eminem::Format::COORDINATE) {
175 // Automatically choosing a smaller integer type for the temporary index.

177 auto primary = (row_ ? NR : NC);

179 if (primary <= limit8) {
