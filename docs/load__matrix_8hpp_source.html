65std::shared_ptr<tatami::Matrix<Value_, Index_> > load_sparse_matrix_basic(Parser_& parser, eminem::Field field, Index_ NR, Index_ NC, eminem::LineIndex NL, bool row) {
66 std::vector<TempIndex_> primary;
67 primary.reserve(NL);
68 std::vector<StoredIndex_> secondary;
69 secondary.reserve(NL);
70 std::vector<StoredValue_> values;
71 values.reserve(NL);
72
73 if (field == eminem::Field::INTEGER) {
74 typedef typename std::conditional<std::is_integral<StoredValue_>::value, StoredValue_, int>::type ParseType;
75 parser.template scan_integer<ParseType>([&](Index_ r, Index_ c, ParseType v) -> void {
76 if (row) {
77 values.push_back(v);
78 primary.push_back(r - 1);
79 secondary.push_back(c - 1);
80 } else {
81 values.push_back(v);
82 primary.push_back(c - 1);
83 secondary.push_back(r - 1);
84 }
85 });
86
87 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
88 typedef typename std::conditional<std::is_floating_point<StoredValue_>::value, StoredValue_, double>::type ParseType;
89 parser.template scan_real<ParseType>([&](Index_ r, Index_ c, ParseType v) -> void {
90 if (row) {
91 values.push_back(v);
92 primary.push_back(r - 1);
93 secondary.push_back(c - 1);
94 } else {
95 values.push_back(v);
96 primary.push_back(c - 1);
97 secondary.push_back(r - 1);
98 }
99 });
100
101 } else {
102 throw std::runtime_error("unsupported Matrix Market field type");
103 }
104
105 auto indptr = tatami::compress_sparse_triplets((row ? NR : NC), values, primary, secondary);
106 return std::shared_ptr<tatami::Matrix<Value_, Index_> >(
107 new tatami::CompressedSparseMatrix<Value_, Index_, decltype(values), decltype(secondary), decltype(indptr)>(
108 NR, NC, std::move(values), std::move(secondary), std::move(indptr), row, false
109 )
110 );

148template<typename Value_, typename Index_, typename StoredValue_, typename Parser_>
149std::shared_ptr<tatami::Matrix<Value_, Index_> > load_dense_matrix_basic(Parser_& parser, eminem::Field field, Index_ NR, Index_ NC, bool row) {
150 std::vector<StoredValue_> values;
151 auto full_size = sanisizer::product<decltype(values.size())>(NR, NC);
152 if (row) {
153 values.resize(full_size);
154 } else {
155 values.reserve(full_size);
156 }
157
158 if (field == eminem::Field::INTEGER) {
159 typedef typename std::conditional<std::is_integral<StoredValue_>::value, StoredValue_, int>::type ParseType;
160 parser.template scan_integer<ParseType>([&](Index_ r, Index_ c, ParseType v) -> void {
161 if (row) {
162 values[sanisizer::nd_offset<decltype(values.size())>(c - 1, NC, r - 1)] = v;
163 } else {
164 values.push_back(v); // Matrix Market ARRAY format is already column-major
165 }
166 });
167
168 } else if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
169 typedef typename std::conditional<std::is_floating_point<StoredValue_>::value, StoredValue_, double>::type ParseType;
170 parser.template scan_real<ParseType>([&](Index_ r, Index_ c, ParseType v) -> void {
171 if (row) {
172 values[sanisizer::nd_offset<decltype(values.size())>(c - 1, NC, r - 1)] = v;
173 } else {
174 values.push_back(v);
175 }
176 });
177
178 } else {
179 throw std::runtime_error("unsupported Matrix Market field type");
180 }
181
182 return std::shared_ptr<tatami::Matrix<Value_, Index_> >(
183 new tatami::DenseMatrix<Value_, Index_, decltype(values)>(NR, NC, std::move(values), row)
184 );
185}

209template<typename Value_, typename Index_, typename StoredValue_ = Automatic, typename StoredIndex_ = Automatic>
210std::shared_ptr<tatami::Matrix<Value_, Index_> > load_matrix(byteme::Reader& reader, const Options& options) {
211 byteme::PerByteSerial<char, byteme::Reader*> pb(&reader);
212 eminem::Parser<decltype(&pb), Index_> parser(&pb, [&]{
213 eminem::ParserOptions eopt;
214 eopt.num_threads = options.num_threads;
215 return eopt;
216 }());
217
218 parser.scan_preamble();
219 const auto& banner = parser.get_banner();
220 auto field = banner.field;
221 auto format = banner.format;
222 auto NR = parser.get_nrows(), NC = parser.get_ncols();
223 auto NL = parser.get_nlines();
224
225 if (format == eminem::Format::COORDINATE) {
226 // Automatically choosing a smaller integer type for the temporary index.
227 constexpr auto limit8 = std::numeric_limits<uint8_t>::max();
228 constexpr auto limit16 = std::numeric_limits<std::uint16_t>::max();
229 auto primary = (options.row ? NR : NC);
230
231 if (sanisizer::is_less_than_or_equal(primary, limit8)) {
232 return internal::load_sparse_matrix_index<Value_, Index_, StoredValue_, StoredIndex_, std::uint8_t>(parser, field, NR, NC, NL, options.row);
233 } else if (sanisizer::is_less_than_or_equal(primary, limit16)) {
234 return internal::load_sparse_matrix_index<Value_, Index_, StoredValue_, StoredIndex_, std::uint16_t>(parser, field, NR, NC, NL, options.row);
235 } else {
236 return internal::load_sparse_matrix_index<Value_, Index_, StoredValue_, StoredIndex_, std::uint32_t>(parser, field, NR, NC, NL, options.row);
237 }
238
239 } else {
240 if constexpr(std::is_same<StoredValue_, Automatic>::value) {
241 if (field == eminem::Field::REAL || field == eminem::Field::DOUBLE) {
242 return internal::load_dense_matrix_basic<Value_, Index_, double>(parser, field, NR, NC, options.row);
243 }
244 if (field != eminem::Field::INTEGER) {
245 throw std::runtime_error("unsupported Matrix Market field type");
246 }
247 return internal::load_dense_matrix_basic<Value_, Index_, int>(parser, field, NR, NC, options.row);
248
249 } else {
250 return internal::load_dense_matrix_basic<Value_, Index_, StoredValue_>(parser, field, NR, NC, options.row);
251 }
252 }
